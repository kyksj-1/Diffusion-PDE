\section{Related Work}

% ----- (a) Function-space diffusion -----
\todo{Group (a): Function-space diffusion solvers — DDO \citep{lim2023ddo}, DiffusionPDE \citep{huang2024diffusionpde}, FunDPS \citep{yao2025fundps}. Acknowledge their Banach-space framework and SOTA on smooth PDEs.}

% ----- (b) Flow matching for PDE -----
\todo{Group (b): Flow matching variants — Stochastic Interpolants \citep{albergo2024interpolants}, Functional Mean Flow \citep{kerrigan2024functionalmf}, CFO \citep{cfo2025}, UniFluids \citep{unifluids2025}. Position our work as orthogonal: structurally aware of shocks, not just architectural variants of flow matching.}

% ----- (c) PDE perspective on diffusion -----
\todo{Group (c): The PDE viewpoint on diffusion models — Score Shocks \citep{sarkar2026scoreshocks} (the theoretical key for this paper), "PDE perspective on generative diffusion" \citep{pdeperspective2025}.}

% ----- (d) Hyperbolic numerics -----
\todo{Group (d): Classical numerics for hyperbolic conservation laws — Godunov, WENO5 \citep{shu1999weno}, Discontinuous Galerkin \citep{cockburn2001dg}. We use these solely as ground-truth generators and as comparators in the entropy-solution sense.}
